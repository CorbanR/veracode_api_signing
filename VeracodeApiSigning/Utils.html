def get_path_and_params_from_url(url)
  uri = parsed_url(url)
  path = uri.path
  params = uri.query
  return "" if (path.nil? || path.empty?) && params.nil?

  built_url = URI::HTTPS.build(path: path, query: params)
  built_url.request_uri
end
